CREATE TABLE COUNTRY (
    country_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    area INT
);

CREATE TABLE CITY (
    city_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    population INT,
    country_id INT NOT NULL,
    CONSTRAINT fk_city_country FOREIGN KEY (country_id) REFERENCES COUNTRY(country_id)
);

CREATE TABLE OWNER (
    owner_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    city_id INT NOT NULL,
    CONSTRAINT fk_owner_city FOREIGN KEY (city_id) REFERENCES CITY(city_id)
);

CREATE TABLE PIZZERIA (
    pizzeria_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    owner_id INT NOT NULL UNIQUE,
    CONSTRAINT fk_pizzeria_owner FOREIGN KEY (owner_id) REFERENCES OWNER(owner_id)
);

CREATE TABLE MENU (
    menu_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    pizzeria_id INT NOT NULL,
    CONSTRAINT fk_menu_pizzeria FOREIGN KEY (pizzeria_id) REFERENCES PIZZERIA(pizzeria_id)
);

CREATE TABLE PLATE (
    plate_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    price NUMBER(10,2) NOT NULL,
    weight INT,
    menu_id INT NOT NULL,
    CONSTRAINT fk_plate_menu FOREIGN KEY (menu_id) REFERENCES MENU(menu_id)
);

CREATE TABLE DEPARTMENT (
    department_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    pizzeria_id INT NOT NULL,
    CONSTRAINT fk_department_pizzeria FOREIGN KEY (pizzeria_id) REFERENCES PIZZERIA(pizzeria_id)
);

CREATE TABLE TEAM (
    team_id INT PRIMARY KEY,
    role VARCHAR(100) NOT NULL
);

CREATE TABLE DEPARTMENT_TEAM_MAPPING (
    department_id INT,
    team_id INT,
    CONSTRAINT pk_dept_team PRIMARY KEY (department_id, team_id),
    CONSTRAINT fk_mapping_department FOREIGN KEY (department_id) REFERENCES DEPARTMENT(department_id),
    CONSTRAINT fk_mapping_team FOREIGN KEY (team_id) REFERENCES TEAM(team_id)
);

CREATE TABLE EMPLOYEE (
    employee_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    age INT,
    gender VARCHAR(1) CHECK (gender IN ('M', 'F')),
    team_id INT NOT NULL,
    CONSTRAINT fk_employee_team FOREIGN KEY (team_id) REFERENCES TEAM(team_id)
);

CREATE SEQUENCE seq_country START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE seq_city START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE seq_owner START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE seq_pizzeria START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE seq_menu START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE seq_plate START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE seq_department START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE seq_team START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE seq_employee START WITH 1 INCREMENT BY 1;
